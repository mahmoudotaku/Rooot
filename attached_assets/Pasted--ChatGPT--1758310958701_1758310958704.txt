import zipfile
import os

# إنشاء مجلد مؤقت
base_dir = "/mnt/data/seo_fix"
os.makedirs(base_dir, exist_ok=True)

# 1. robots.txt
robots_content = """User-agent: *
Allow: /

Sitemap: https://rooot-production.up.railway.app/sitemap.xml
"""
with open(os.path.join(base_dir, "robots.txt"), "w") as f:
    f.write(robots_content)

# 2. sitemap.xml
sitemap_content = """<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  <url>
    <loc>https://rooot-production.up.railway.app/</loc>
    <lastmod>2025-09-19</lastmod>
    <priority>1.0</priority>
  </url>
  <url>
    <loc>https://rooot-production.up.railway.app/about</loc>
    <lastmod>2025-09-19</lastmod>
    <priority>0.8</priority>
  </url>
  <url>
    <loc>https://rooot-production.up.railway.app/contact</loc>
    <lastmod>2025-09-19</lastmod>
    <priority>0.8</priority>
  </url>
</urlset>
"""
with open(os.path.join(base_dir, "sitemap.xml"), "w") as f:
    f.write(sitemap_content)

# 3. مثال صفحة HTML محسنة
html_content = """<!DOCTYPE html>
<html lang="ar">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Rooot Terminal - تعلم أوامر لينكس وأدوات الأمن السيبراني أونلاين</title>
  <meta name="description" content="Rooot Terminal أداة مجانية تحاكي أوامر لينكس وأدوات الأمن السيبراني أونلاين للتعلم والتجربة بدون الحاجة لتثبيت برامج.">
  <link rel="canonical" href="https://rooot-production.up.railway.app/" />
</head>
<body>
  <header>
    <h1>Rooot Terminal - محاكي أوامر لينكس أونلاين</h1>
  </header>

  <main>
    <h2>تعلم الأمن السيبراني وأوامر لينكس بسهولة</h2>
    <p>جرب أوامر مثل <strong>ping</strong>, <strong>nmap</strong>, <strong>sqlmap</strong> وغيرهم مباشرة من المتصفح.</p>

    <img src="logo.png" alt="شعار Rooot Terminal" width="200">
  </main>

  <footer>
    <p>© 2025 Rooot Terminal. جميع الحقوق محفوظة.</p>
  </footer>
</body>
</html>
"""
with open(os.path.join(base_dir, "index.html"), "w") as f:
    f.write(html_content)

# ضغط الملفات في zip
zip_path = "/mnt/data/seo_fix_files.zip"
with zipfile.ZipFile(zip_path, 'w') as zipf:
    for file_name in os.listdir(base_dir):
        zipf.write(os.path.join(base_dir, file_name), file_name)
